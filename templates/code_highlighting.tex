import simpy

def example(env):
    yield env.timeout(10)
    print("now=%d" % env.now)
    yield env.timeout(10)
    print("now=%d" % env.now)

env = simpy.Environment()
p = env.process(example(env))
env.run()

# Output:
# now=10
# now=20
